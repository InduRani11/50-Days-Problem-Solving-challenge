/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {

        ArrayList<Integer> ar=new ArrayList<>();
        if(root==null){
            return ar;
        }
        ar.add(root.val);
        ar.addAll(preorderTraversal(root.left));
        ar.addAll(preorderTraversal(root.right));
        return ar;
    }
}
